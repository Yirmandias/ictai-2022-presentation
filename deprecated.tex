\textbf{Operational model:}
\tiny
\begin{lstlisting}(do
(def-method m1
'((:task pick-and-drop)
  (:params (?ball ball)
    (?room room)
    (?gripper gripper)
    (?departure room))
  (:pre-conditions (and-cond 
    ( = (at ?ball) (at-robby))
    (= (carry ?gripper) no_ball)
    (= ?departure (at-robby))))
  (:body
    (do
     (pick ?ball ?departure ?gripper)
     (t_move ?room)
     (drop ?ball ?room ?gripper)))))
\end{lstlisting}

(def-action-model move
'((:params (?from room) (?to room))
  (:pre-conditions (and-cond 
    (= (at-robby) ?from)
    (!= ?from ?to))
    (= (connected ?from ?to) yes))
  (:effects
    (begin
     (assert 'at-robby ?to)))))
\end{lstlisting}
